### Example 01: Basic Document Creation

**File:** `01_basic_document.py`

```python
#!/usr/bin/env python3
"""
DOCX Example 01: Basic Document Creation

Creates a simple Word document with headings and paragraphs.
"""

from docx import Document
from docx.shared import Pt, Inches

def create_basic_document():
    doc = Document()

    # Title
    doc.add_heading('Document Processing Report', 0)

    # Subtitle
    doc.add_paragraph('Generated by Kilo Agent')
    doc.paragraphs[-1].alignment = 1  # Center

    # Section 1
    doc.add_heading('Introduction', level=1)
    doc.add_paragraph(
        'This report demonstrates basic document creation capabilities. '
        'It includes various formatting options and structure elements.'
    )

    # Section 2
    doc.add_heading('Key Findings', level=1)
    findings = [
        'Document automation increases productivity by 40%',
        'Consistent formatting improves readability',
        'Template-based creation ensures brand compliance'
    ]
    for finding in findings:
        doc.add_paragraph(finding, style='List Bullet')

    # Section 3
    doc.add_heading('Recommendations', level=1)
    doc.add_paragraph(
        'Based on our analysis, we recommend implementing automated '
        'document generation for all standard report types.'
    )

    return doc

if __name__ == "__main__":
    doc = create_basic_document()
    output_path = "example_basic.docx"
    doc.save(output_path)
    print(f"[OK] Created: {output_path}")
```

### Example 02: Document with Tables

**File:** `02_document_with_tables.py`

```python
#!/usr/bin/env python3
"""
DOCX Example 02: Document with Tables

Creates a document containing formatted tables.
"""

from docx import Document
from docx.shared import Pt, RGBColor
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.enum.table import WD_TABLE_ALIGNMENT

def create_document_with_tables():
    doc = Document()
    doc.add_heading('Sales Report with Tables', 0)

    # Summary table
    doc.add_heading('Quarterly Summary', level=1)
    table = doc.add_table(rows=5, cols=3)
    table.style = 'Table Grid'

    # Header row
    headers = ['Quarter', 'Revenue', 'Growth']
    for i, header in enumerate(headers):
        cell = table.rows[0].cells[i]
        cell.text = header
        cell.paragraphs[0].runs[0].bold = True

    # Data rows
    data = [
        ('Q1 2024', '$1.2M', '+15%'),
        ('Q2 2024', '$1.5M', '+25%'),
        ('Q3 2024', '$1.8M', '+20%'),
        ('Q4 2024', '$2.1M', '+17%')
    ]
    for row_idx, (quarter, revenue, growth) in enumerate(data, start=1):
        table.rows[row_idx].cells[0].text = quarter
        table.rows[row_idx].cells[1].text = revenue
        table.rows[row_idx].cells[2].text = growth

    # Details table
    doc.add_heading('Regional Breakdown', level=1)
    table2 = doc.add_table(rows=4, cols=3)
    table2.style = 'Table Grid'

    headers2 = ['Region', 'Sales', '% of Total']
    for i, header in enumerate(headers2):
        table2.rows[0].cells[i].text = header
        table2.rows[0].cells[i].paragraphs[0].runs[0].bold = True

    regions = [
        ('North America', '$3.2M', '38%'),
        ('Europe', '$2.8M', '33%'),
        ('Asia Pacific', '$2.4M', '29%')
    ]
    for row_idx, (region, sales, pct) in enumerate(regions, start=1):
        table2.rows[row_idx].cells[0].text = region
        table2.rows[row_idx].cells[1].text = sales
        table2.rows[row_idx].cells[2].text = pct

    return doc

if __name__ == "__main__":
    doc = create_document_with_tables()
    doc.save("example_tables.docx")
    print("[OK] Created: example_tables.docx")
```

### Example 03: Document with Styles

**File:** `03_document_with_styles.py`

```python
#!/usr/bin/env python3
"""
DOCX Example 03: Document with Styles

Demonstrates applying styles and formatting.
"""

from docx import Document
from docx.shared import Pt, Inches, RGBColor
from docx.enum.text import WD_ALIGN_PARAGRAPH

def create_styled_document():
    doc = Document()

    # Custom title
    title = doc.add_heading('', 0)
    run = title.add_run('Styled Document Report')
    run.font.size = Pt(28)
    run.font.bold = True
    run.font.color.rgb = RGBColor(0, 51, 102)

    doc.add_paragraph('')  # Spacer

    # Styled paragraphs
    p = doc.add_paragraph()
    run = p.add_run('Important: ')
    run.bold = True
    run.font.color.rgb = RGBColor(200, 0, 0)
    run = p.add_run('This section contains critical information that requires attention.')
    run.font.size = Pt(12)

    # Highlighted text
    p2 = doc.add_paragraph()
    run = p2.add_run('Key metrics demonstrate significant improvement in all areas.')
    run.font.size = Pt(14)
    run.font.italic = True

    # Inline formatting demo
    doc.add_paragraph('')
    p3 = doc.add_paragraph('Inline formatting example: ')
    run1 = p3.add_run('bold')
    run1.bold = True
    run2 = p3.add_run(', ')
    run3 = p3.add_run('italic')
    run3.italic = True
    run4 = p3.add_run(', ')
    run5 = p3.add_run('underlined')
    run5.underline = True

    return doc

if __name__ == "__main__":
    doc = create_styled_document()
    doc.save("example_styled.docx")
    print("[OK] Created: example_styled.docx")
```

### Example 04: Read and Modify Existing Document

**File:** `04_read_existing_document.py`

```python
#!/usr/bin/env python3
"""
DOCX Example 04: Read and Modify Existing Document

Shows how to load, inspect, and modify existing documents.
"""

from docx import Document

def read_and_modify():
    # First create a sample document to modify
    doc = Document()
    doc.add_heading('Original Document', 0)
    doc.add_paragraph('This is the original content.')
    doc.add_paragraph('Template placeholder: {{CUSTOMER_NAME}}')
    doc.save('sample_to_modify.docx')
    print("[OK] Created sample document")

    # Now read and modify it
    doc = Document('sample_to_modify.docx')

    # Inspect content
    print(f"Paragraphs: {len(doc.paragraphs)}")
    print(f"Tables: {len(doc.tables)}")

    # Replace placeholders
    for para in doc.paragraphs:
        if '{{CUSTOMER_NAME}}' in para.text:
            para.text = para.text.replace('{{CUSTOMER_NAME}}', 'Acme Corporation')

    doc.save('sample_modified.docx')
    print("[OK] Created modified document")
    return doc

if __name__ == "__main__":
    read_and_modify()
```

### Example 05: Extract Text from Document

**File:** `05_extract_text.py`

```python
#!/usr/bin/env python3
"""
DOCX Example 05: Extract Text from Document

Shows how to extract and process text content from DOCX files.
"""

from docx import Document

def extract_text():
    # Create sample document with various content
    doc = Document()
    doc.add_heading('Sample Document', 0)
    doc.add_paragraph('First paragraph of text.')
    doc.add_paragraph('Second paragraph with more content.')

    table = doc.add_table(rows=2, cols=2)
    table.cell(0, 0).text = 'Header 1'
    table.cell(0, 1).text = 'Header 2'
    table.cell(1, 0).text = 'Data 1'
    table.cell(1, 1).text = 'Data 2'

    doc.save('sample_for_extraction.docx')

    # Extract text
    doc = Document('sample_for_extraction.docx')

    print("=== Document Text ===")
    for i, para in enumerate(doc.paragraphs):
        print(f"Para {i}: {para.text}")

    print("\n=== Table Data ===")
    for table in doc.tables:
        for row in table.rows:
            row_data = [cell.text for cell in row.cells]
            print(' | '.join(row_data))

if __name__ == "__main__":
    extract_text()
```

## Batch Processing

For batch processing multiple documents, use:

```bash
# Convert all documents in a folder to text
for file in *.docx; do
    python -c "from docx import Document; d=Document('$file'); print('\\n'.join([p.text for p in d.paragraphs]))" > "${file%.docx}.txt"
done
```
